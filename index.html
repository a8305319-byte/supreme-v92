import time import random class GodSystemAI: def __init__(self): # 狀態機：0=面試, 1=入職後 self.state = 0 self.driver_name = "大哥" # 隨機分配入職後的角色 self.support_roles = [ {"name": "阿翔", "style": "大哥感", "tone": "台式口吻"}, {"name": "小潔", "style": "行政感", "tone": "溫柔專業"} ] self.assigned_role = random.choice(self.support_roles) def human_delay(self, text): """模擬真人打字延遲，不秒回""" wait = len(text) * 0.05 + random.uniform(1.0, 2.0) time.sleep(wait) def process_input(self, user_text): # 簡單的台語關鍵字識別邏輯 is_taiwanese = any(word in user_text for word in ["好勢", "會使", "無代誌", "食飽", "可以啦"]) if self.state == 0: return self.interview_mode(user_text, is_taiwanese) else: return self.support_mode(user_text, is_taiwanese) def interview_mode(self, user_text, is_taiwanese): """面試引路人模式：專業、引導、非考官""" self.human_delay("正在處理面試回覆...") # 模擬面試題目邏輯 if "準備好了" in user_text or "好勢" in user_text: response = "收到，專業。第一題：如果配送時客戶沒接電話，現場又不能違停，你會怎麼處理？" elif "拍照" in user_text or "回報" in user_text: response = "好的，這個邏輯很清楚，我們記錄下來了。面試到此結束，請稍候，系統正在核對資料..." self.state = 1 # 自動轉入職狀態 else: response = "了解，大哥你照經驗說就好，我這邊都會記錄。" return f"【AI 面試引路人】: {response}" def support_mode(self, user_text, is_taiwanese): """司機端客服模式：擬人、隨機人設、不廢話""" role_name = self.assigned_role['name'] self.human_delay(user_text) # 根據角色風格生成回覆 if self.assigned_role['name'] == "阿翔": prefix = "嘿呀，" if not is_taiwanese else "好勢啦，" responses = [ f"{prefix}恭喜喔！資料過件了。我是阿翔，以後後台我負責，多照顧啦！", f"{prefix}大哥，我看你資料入庫了，那個薪資條記得橫著螢幕簽名一下嘿。", f"{prefix}沒事啦，以後有問題直接敲我，我都在。" ] else: # 小潔 prefix = "大哥～" responses = [ f"{prefix}恭喜你入職唷！我是小潔，以後有對帳問題都可以找我呀。", f"{prefix}幫你確認好入職流程了，再麻煩幫我補個簽名唷，謝謝辛苦了！" ] return f"【{role_name}】: {random.choice(responses)}" # --- 模擬運行 --- sys = GodSystemAI() print(">>> 系統啟動：進入面試模式") print("【AI 面試引路人】: 大哥你好，歡迎來面試。準備好我們就開始？") # 模擬對話流程 actions = [ "好勢，隨時可以開始。", "我會先拍照，然後回報後台處理。", "阿翔哥多照顧！" ] for action in actions: print(f"\n司機回覆: {action}") result = sys.process_input(action) print(result)

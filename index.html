// 在提交申請的 function 中加入檢查邏輯 window.submitApply = async () => { const pin = document.getElementById('app_pin').value; // 🔍 檢查 PIN 碼是否已經被佔用 (包括審核中與已核准的) const isDuplicate = partners.some(p => p.uid === pin); if (isDuplicate) { alert("❌ 此 PIN 碼已被其他司機使用，請重新設定（建議使用手機末 4 碼）"); document.getElementById('app_pin').value = ""; return; } const btn = document.getElementById('subBtn'); btn.innerHTML = "提交中..."; btn.disabled = true; const data = { name: document.getElementById('app_name').value, uid: pin, // ...其餘欄位保持不變 }; try { await addDoc(collection(db, 'partners'), data); alert("✅ 申請成功！請靜候審核。"); location.reload(); } catch (e) { alert("提交失敗"); btn.disabled = false; } };
